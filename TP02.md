## Exercice 1. Feu de circulation
``` from microbit import *
color = 1

while True:

    if (color == 1):
        pin0.write_digital(1)
        pin16.write_digital(0)
        pin8.write_digital(0)
        color = 2
    elif (color == 2):
        pin8.write_digital(1)
        pin16.write_digital(0)
        pin0.write_digital(0)
        color = 3
    elif color == 3:
        pin8.write_digital(0)
        pin16.write_digital(1)
        pin0.write_digital(0)
        color = 1
    else:
        pin0.write_digital(1)
        pin8.write_digital(1)
        pin16.write_digital(1)
    sleep(2000) 
    ```
